    <p className="text-lg text-gray-600 dark:text-gray-400 mb-8">Get building in minutes with any supported language</p>
    
    <Steps>
      <Step title="Add build configuration to pixi.toml">
        Specify which backend to use for your project:
        
        ```toml pixi.toml
        [package]
        name = "my-package"
        version = "0.1.0"

        [package.build]
        backend = { name = "pixi-build-python", version = "*" }
        channels = ["https://prefix.dev/conda-forge"]
        ```
      </Step>
      
      <Step title="Build your package">
        Run the build command to create a conda package:
        
        ```bash
        pixi build
        ```
        
        <Note>
          The backend automatically generates a complete rattler-build recipe and handles all build dependencies.
        </Note>
      </Step>
      
      <Step title="Install or distribute">
        Install locally or publish to a conda channel:
        
        ```bash
        # Install locally
        pixi install
        
        # Or upload to a channel
        pixi upload output/my-package-0.1.0-*.conda
        ```
      </Step>
    </Steps>
  </div>
